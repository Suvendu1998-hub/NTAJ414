CREATE TABLE "SYSTEM"."PERSON_PROFILE" 
   (	"PID" NUMBER(10,0) NOT NULL ENABLE, 
	"PNAME" VARCHAR2(20 BYTE), 
	"PADDRS" VARCHAR2(20 BYTE), 
	"RESUME" VARCHAR2(50 BYTE), 
	"PHOTO" VARCHAR2(50 BYTE), 
	 CONSTRAINT "PERSON_PROFILE_PK" PRIMARY KEY ("PID"))
	 ======================================
	 CREATE SEQUENCE  "SYSTEM"."PID_SEQUENCE"  MINVALUE 1 MAXVALUE 1000 INCREMENT BY 1 START WITH 1
